@(groupId: String, addDocumentation: Boolean)
package @groupId

import sttp.tapir.server.netty.NettyFutureServer

import scala.concurrent.duration.Duration
import scala.concurrent.{Await, ExecutionContext, Future}
import scala.io.StdIn

object Main {
  implicit val ec = ExecutionContext.global

  def main(args: Array[String]): Unit = {
    val port = sys.env.get("http.port").map(_.toInt).getOrElse(8080)
    val program = for {
      binding <- NettyFutureServer().port(port).addEndpoints(Endpoints.all).start()
      _ <- Future {
        println(s"@if(addDocumentation){Go to http://localhost:${binding.port}/docs to open SwaggerUI.}else{Server started at http://localhost:${binding.port}.} Press ENTER key to exit.")
        StdIn.readLine()
      }
      stop <- binding.stop()
    } yield stop

    Await.result(program, Duration.Inf)
  }
}
